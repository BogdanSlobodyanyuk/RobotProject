use robot_dreams;

CREATE table countries
(
    id   int PRIMARY KEY NOT NULL AUTO_INCREMENT,
    name varchar(40),
    country_population  int
);


CREATE table cities
(
    id   int PRIMARY KEY NOT NULL AUTO_INCREMENT,
    name varchar(40),
    country_id int,
    city_population int

);

alter table cities
    ADD FOREIGN KEY (country_id) REFERENCES countries(id);
